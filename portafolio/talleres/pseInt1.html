	definir nombre, apellidos Como Caracter;

	definir estatura Como Real;

	Limpiar Pantalla;
	escribir "Hola ", nombre;

Realizar un programa el cual solicite su nombre y apellidos, también debe capturar nombre y apellidos de su padre y madre. Al finalizar debe imprimir en pantalla el siguiente mensaje "Yo [Nombre Completo], soy hijo de [Nombre de la Madre] y [Nombre del Padre]. 
Proceso variablesImpresion3
	
	definir nombre, apellidos Como Caracter;
	definir nombreMadre, apellidosMadre Como Caracter;
	definir nombrePadre, apellidosPadre Como Caracter;
	//datos personales
	escribir "Ingrese su nombre:";
	leer nombre;
	escribir "Ingrese sus apellidos:";
	leer apellidos;
	Limpiar Pantalla;
	//datos madre
	escribir "Ingrese el nombre de su madre:";
	leer nombreMadre;
	escribir "Ingrese los apellidos de su madre:";
	leer apellidosMadre;
	Limpiar Pantalla;
	//datos padre
	escribir "Ingrese el nombre de su padre:";
	leer nombrePadre;
	escribir "Ingrese los apellidos de su padre:";
	leer apellidosPadre;
	Limpiar Pantalla;
	
	escribir "Yo ", nombre, " ", apellidos, ", soy hijo de ", nombreMadre, " ", apellidosMadre, " y ", nombrePadre, " ", apellidosPadre;
	
FinProceso

Realizar un programa el cual solicite el nombre de una ciudad capital y el país. Al finalizar debe imprimir en pantalla el siguiente mensaje "La ciudad [Nombre de la Ciudad], es la capital del país [Nombre del País] 
Proceso variablesImpresion4
	
	definir capital, pais Como Caracter;
	
	escribir "Ingrese la ciudad Capital de un país:";
	leer capital;
	escribir "Ahora, ingrese el país respectivo de la capital anterior:";
	leer pais;
	Limpiar Pantalla;
	
	escribir "La ciudad ", capital, ", es la capital del país ", pais;

Realizar un programa el cual solicite el nombre de su mascota, edad de la mascota, el tipo de mascota y su nombre completo. Al finalizar el sistema debe e imprimir en pantalla el siguiente mensaje: [Nombre de Mascota] es un(a) [Tipo de Mascota], el cual, tiene [Edad de la Mascota] años de edad y [Nombre Completo] es actualmente su dueño(a). 
Proceso variablesImpresion5
	
	definir namePet, pet Como Caracter;
	definir agePet Como Entero;
	definir name Como Caracter;
	
	escribir "Ingrese el nombre de su mascota:";
	leer namePet;
	escribir "¿Cuál es la edad de ", namePet, "?";
	leer agePet;
	escribir "¿Cuál animal es ", namePet, "?";
	leer pet;
	escribir "Por último, ingresa tu nombre completo:";
	leer name;
	Limpiar Pantalla;
	
	escribir namePet, " es un ", pet, " el cual, tiene ", agePet, " años de edad y ", name, " es actualmente su dueño(a)";
